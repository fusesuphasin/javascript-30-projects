Common Mouse Event Listeners

            
                const click = document.querySelect("#click");
                click.addEventListener("click", function() {
                    clickCode.style.display = "block";
                });
            
        
            
                const dbclick = document.querySelect("#dbclick");
                dbclick.addEventListener("dbclick", function() {
                    dbclick.style.display = "block";
                });
            
        
            
                const contextmenu = document.querySelect("#contextmenu");
                contextmenu.addEventListener("contextmenu", function(e) {
                    contextmenu.style.display = "block";
                    e.preventDefault();
                });
            
        
            
                const mouseover = document.querySelect("#mousover");
                    mouseover.addEventListener("mouseover", function() {
                    mouseoverCode.style.display = "block";
                });
            
        
            
                const mouseleave = document.querySelect("#mouseleave");
                    mouseleave.addEventListener("mouseleave", function() {
                    mouseleaveCode.style.display = "block";
                });
            
        
            
                const wheel = document.querySelect("#wheel");
                    wheel.addEventListener("wheel", function() {
                    wheelCode.style.display = "block";
                });
            
        
            
                const select = document.querySelector("#select");
                    select.addEventListener("select", function() {
                        selectCode.style.display = "block";
                    });
